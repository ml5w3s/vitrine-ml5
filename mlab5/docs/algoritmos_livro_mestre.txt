class ListaEncadeada:
    def __init__(self):
        self.cabeca = None    # início da lista
    
    def adicionar_no_final(self, valor):
        novo_no = No(valor)
        if self.cabeca is None:
            self.cabeca = novo_no
            return
        
        atual = self.cabeca
        while atual.proximo is not None:
            atual = atual.proximo
        atual.proximo = novo_no
    
    def buscar(self, alvo):
        atual = self.cabeca
        indice = 0
        
        while atual is not None:
            if atual.valor == alvo:
                return indice  # encontrou!
            atual = atual.proximo
            indice += 1
        return -1  # não encontrado
    
    def __str__(self):
        valores = []
        atual = self.cabeca
        while atual is not None:
            valores.append(str(atual.valor))
            atual = atual.proximo
        return " -> ".join(valores) if valores else "Lista vazia"

frutas = ["maçã", "banana", "laranja", "kiwi", "uva"]

def busca_linear(lista, alvo):
    for i in range(len(lista)):
        if lista[i] == alvo:    # comparação de strings com ==
            return i
    return -1

print(busca_linear(frutas, "laranja"))  # → 2
print(busca_linear(frutas, "abacaxi"))   # → -1 (não encontrado)

Título: Busca binária com strings
frutas_ordenadas = ["banana", "kiwi", "laranja", "maçã", "uva"]
def busca_binaria(lista, alvo):
    esquerda = 0
    direita = len(lista) - 1
    while esquerda <= direita:
        meio = (esquerda + direita) // 2
        if lista[meio] == alvo:
            return meio
        elif lista[meio] < alvo:         # comparação lexicográfica (alfabética)
            esquerda = meio + 1
        else:
            direita = meio - 1
    return -1
print(busca_binaria(frutas_ordenadas, "laranja"))  # → 2
print(busca_binaria(frutas_ordenadas, "abacaxi"))  # → -1

const frutas = ["maçã", "banana", "laranja", "kiwi", "uva"];

function buscaLinear(lista, alvo) {
    for (let i = 0; i < lista.length; i++) {
        if (lista[i] === alvo) {  // comparação estrita de strings (recomendado)
            return i;             // retorna o índice da primeira ocorrência
        }
    }
    return -1;  // não encontrado
}

Título quick sort
numeros = [8,1,10,3,55,98,65,9,21,46,61]
def quick_sort(lista):
    if len(lista) <= 1:
        return lista
    pivo = lista[len(lista) // 2]
    esquerda = [x for x in lista if x < pivo]
    meio = [x for x in lista if x == pivo]
    direita = [x for x in lista if x > pivo]
    return quick_sort(esquerda) + meio + quick_sort(direita)
print(quick_sort(numeros))
