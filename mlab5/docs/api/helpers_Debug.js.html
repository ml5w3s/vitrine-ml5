/**
 * @file Debug.js
 * @description Módulo centralizado para logging estratégico em aplicações SPA.
 * Fornece métodos estáticos para logar mensagens, erros e tabelas no console e atualizar uma view de debug opcional.
 * 
 * @example
 * Debug.log('Router', 'Navegação iniciada', { from: '/home', to: '/about' });
 * Debug.error('ApiService', new Error('Falha na requisição'));
 * Debug.table('Store', 'Estado Atual', storeData);
 */
export class Debug {
    /** 
     * @type {boolean}
     * @description Define se os logs estão habilitados globalmente.
     */
    static enabled = true;

    /** 
     * @type {?DebugView}
     * @description Referência opcional para a instância de uma view de debug (UI). 
     */
    static view = null;

    /**
     * Registra uma instância de view que exibirá logs e erros na interface.
     * @param {DebugView} viewInstance - Instância concreta que implementa `addLog`, `addError` e `updateInfo`.
     * @example
     * const view = new DebugView();
     * Debug.registerView(view);
     */
    static registerView(viewInstance) {
        this.view = viewInstance;
    }

    /**
     * Loga uma mensagem agrupada no console.
     * Também envia a mensagem para a view, se existir.
     * @param {string} group - Nome do grupo (ex: 'Router', 'ApiService').
     * @param {...any} args - Mensagens ou objetos a serem logados.
     */
    static log(group, ...args) {
        if (!this.enabled) return;

        console.groupCollapsed(`[${group}] - ${args[0]}`);
        args.slice(1).forEach(arg => console.log(arg));
        console.groupEnd();

        if (this.view) {
            this.view.addLog(`[${group}] ${args[0]}`);
        }
    }

    /**
     * Loga erros agrupados no console e na view.
     * @param {string} group - Nome do grupo.
     * @param {...any} args - Mensagens ou objetos de erro.
     */
    static error(group, ...args) {
        if (!this.enabled) return;

        console.group(`[${group}] - ERRO`);
        args.forEach(arg => console.error(arg));
        console.groupEnd();

        if (this.view) {
            const errorMessage = args.map(arg => arg.message || String(arg)).join(' ');
            this.view.addError(`[${group}] ${errorMessage}`);
        }
    }

    /**
     * Loga dados em formato de tabela no console.
     * @param {string} group - Nome do grupo.
     * @param {string} title - Título da tabela.
     * @param {object|Array} data - Dados a serem exibidos em formato tabular.
     */
    static table(group, title, data) {
        if (!this.enabled) return;

        console.groupCollapsed(`[${group}] - Tabela: ${title}`);
        console.table(data);
        console.groupEnd();
    }

    /**
     * Atualiza uma informação específica na view de debug.
     * @param {string} key - Nome da chave (ex: 'route', 'user').
     * @param {string} value - Valor a ser exibido.
     */
    static updateView(key, value) {
        if (this.view) {
            this.view.updateInfo(key, value);
        }
    }
}
